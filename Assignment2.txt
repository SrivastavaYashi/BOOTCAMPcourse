7. Write a program to find the position of first 1 in LSB.
#include<stdio.h>
int main()
{
 int n,y,count=1;
 printf("Enter a number=");
 scanf("%d",&n);
   while(1)
  {
    y=n&1;
    if(y==0)
    {
    n=n>>1;
    count++;
    }
    else
    break;
  }
    printf("%d",count);
}
